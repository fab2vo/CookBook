10   public abstract class SingleFragmentActivity extends AppCompatActivity {

14           super.onCreate(savedInstanceState);
15           setContentView(R.layout.activity_fragment);
16           FragmentManager fm = getSupportFragmentManager();
17           Fragment fragment = fm.findFragmentById(R.id.fragment_container);
18           if (fragment == null) {
19               fragment = createFragment();
20               fm.beginTransaction()
21                       .add(R.id.fragment_container, fragment)
22                       .commit();
23           }
24       }
