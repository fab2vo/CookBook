238      class getShortRecipesFromCommunity extends AsyncTask<Integer, Integer, Boolean> {
239          private static final String PHP204 = "return204.php";
240          private static final String PHPGETSHORTRECIPES = "getcommunitynews.php";
241  
242          @Override
243          protected void onPreExecute() {
244              super.onPreExecute();
245              mCookbookShort.getDLStarted();
246              mSelectType=mCookbookShort.getSelectType();
247          }
248  
249          @Override
250          protected void onPostExecute(Boolean b) {
251              super.onPostExecute(b);
252              mCookbookShort.setDLCompleted();
253          }
254  
255          @Override
256          protected Boolean doInBackground(Integer ... integers) {
257              Integer ntoDownload=integers[0];
258              if (!test204()) {deBugShow("204!"); return false;}
259              // download texts
260              String s=getStringRecipesFromServer(0,ntoDownload,mSelectType, false);
261              if (s.equals("")) {deBugShow("Echec query 1"); return false;}
262              if (!fillInitialTable(s)) {deBugShow("Echec parse >"+s+"<"); return false;}
263              publishProgress(1);
264              // download photo i
265              for(Integer i=0;i<mCookbookShort.getsize();i++) {
266                  if(isCancelled()) {deBugShow("Interrupted)"); return false;}
267                  s = getStringRecipesFromServer(i, 1, mSelectType, true);
268                  if (s.equals("")) {
269                      deBugShow("Echec query "+i);
270                      break;
271                  }
272                  if (updatePhotoOfRecipe(s)) publishProgress(i+2);
273              }
274              return true;
275          }
276  
277          @Override
278          protected void onProgressUpdate(Integer ... values){
279              updateUI();
280          }

282          private String getStringRecipesFromServer(Integer start, Integer count, Integer type, Boolean withphoto){
283              HashMap<String, String> data = new HashMap<>();
284              data.put("iduser", mSession.getUser().getId().toString());
285              String comtype="";
286              if (type==RECENT) comtype="RECENT";
287              if (type==POPULAR) comtype="POPULAR";
288              if (type==BESTNOTE) comtype="BESTNOTE";
289              data.put("comtype", comtype);
290              DecimalFormat formatter = new DecimalFormat("00");
291              data.put("start", formatter.format( start));
292              data.put("count", formatter.format(count));
293              data.put("comparam", (withphoto ? "WP":"WOP"));
294              NetworkUtils mNetUtils = new NetworkUtils(getContext());
295              String result = mNetUtils.sendPostRequestJson(mSession.getURLPath() + PHPGETSHORTRECIPES, data);
296              if (result==null) return "";
297              return result;
298          }

300          private boolean fillInitialTable(String s){
301              NetworkUtils mNetUtils = new NetworkUtils(getContext());
302              List<Recipe> recipes=mNetUtils.parseRecipesOfCommunity(s,false);
303              if ((recipes!=null)&&(!recipes.isEmpty())){
304                  mCookbookShort.clear();
305                  mCookbookShort.fill(recipes);
306                  return true;
307              } else return false;
308          }
309  
310          private boolean updatePhotoOfRecipe(String s){
311              NetworkUtils mNetUtils = new NetworkUtils(getContext());
312              List<Recipe> recipes=mNetUtils.parseRecipesOfCommunity(s,true);
313              if ((recipes!=null)&&(!recipes.isEmpty())){
314                  mCookbookShort.updatePhoto(recipes.get(0));
315                  return true;
316              } return false;
317          }

320          private Boolean test204() {
321              try {
322                  URL url = new URL(mSession.getURLPath() + PHP204);
323                  HttpURLConnection conn = (HttpURLConnection) url.openConnection();
324                  conn.setConnectTimeout(mSession.getConnectTimeout());
325                  conn.setReadTimeout(mSession.getReadTimeout());
326                  conn.setRequestMethod("HEAD");
327                  InputStream in = conn.getInputStream();
328                  int status = conn.getResponseCode();
329                  in.close();
330                  conn.disconnect();
331                  return (status == HttpURLConnection.HTTP_NO_CONTENT);
332              } catch (Exception e) {
333                  deBugShow("Test 204 : " + e);
334                  return false;
335              }
336          }
